\text{ of possible permutations}) \\ & \quad={1\over \sqrt{N!}} \sum_{i=1}^{N!} (-1)^{P(i_1,i_2,...i_N)} \phi_{i_1}(x_1)\phi_{i_2}(x_2)...\phi_{i_N}(x_N) \\ & \boxed{\text{single particle density matrix}} \\ & \text{there are }N \text{ single particles} \\ & n(x;y)=N\int dx_2...dx_N \psi(x,x_2...x_N)\psi^*(y,x_2...x_N) \\ & \quad =N\int dx_2...dx_N {1\over \sqrt{N!}} \sum_{i=1}^{N!} (-1)^{P(i_1,i_2,...i_N)} \phi_{i_1}(x)\phi_{i_2}(x_2)...\phi_{i_N}(x_N) \\ & \qquad \qquad \qquad \qquad \quad{1\over \sqrt{N!}} \sum_{k=1}^{N!} (-1)^{P(k_1,k_2,...k_N)} \phi^*_{k_1}(y)\phi^*_{k_2}(x_2)...\phi^*_{k_N}(x_N)\\ & \quad (\text{integral survives iff } i_2...i_N=k_2...k_N)\\ & \quad (\text{and }i_2...i_N=k_2...k_N \Rightarrow i_1=k_1)\\ & \quad ={1\over (N-1)!}\int dx_2...dx_N \sum_{i=1}^{N!} \phi_{i_1}(x)\phi_{i_2}(x_2)...\phi_{i_N}(x_N) \\ & \qquad \qquad \qquad \qquad \qquad \qquad \qquad \phi^*_{i_1}(y)\phi^*_{i_2}(x_2)...\phi^*_{i_N}(x_N)\\ & \quad ={1\over (N-1)!} \sum_{i_1=1}^{N} \phi_{i_1}(x)\phi^*_{i_1}(y) (N-1)!
