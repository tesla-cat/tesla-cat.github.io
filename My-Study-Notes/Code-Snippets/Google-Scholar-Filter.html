Google Scholar Filter¶

Date

1 Mar 2019

  • code

from urllib.request import urlopen
from bs4 import BeautifulSoup

def search(myfile,filters):
    soup = BeautifulSoup(myfile)
    for row in soup('table', {'id': 'gsc_a_t'})[0].tbody('tr'):
        #print('\n', row)
        result = {}
        result['year'] = row('span',{'class':'gsc_a_h gsc_a_hc gs_ibl'})[0].get_text()
        result['publisher'] = row('div',{'class':'gs_gray'})[1].get_text().lower()
        result['title'] = row('a',{'class':'gsc_a_at'})[0].get_text()
        for key in filters:
            if key in result:
                for text in filters[key]:
                    if text.lower() in result[key]:
                        print(result)
